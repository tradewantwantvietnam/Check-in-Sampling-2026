<script>

  // Đã cập nhật chính xác Link Web App Apps Script bạn gửi

  const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbyQ7Pb1Oyvct9bENwOuCyna5qnzaZCavVfFvLNOndYH_I0za4oPqd833mUomsKFBGQy9w/exec';

  const FORM_URL   = 'https://forms.gle/Gyu1uUKSNNDMmSzi7';

  const ENTRY_ID   = 'entry.1519431587';


  let stream = null;

  let capturedDataUrl = null;

  let uploadedLink = null;


  function setStatus(msg, type='') {

    const el = document.getElementById('statusMsg');

    el.textContent = msg;

    el.className = 'status ' + type;

  }


  function updateSteps(active) {

    [1,2,3].forEach(i => {

      const dot = document.getElementById('dot'+i);

      dot.className = 'step-dot' + (i < active ? ' done' : i === active ? ' active' : '');

    });

    [1,2].forEach(i => {

      const line = document.getElementById('line'+i);

      line.className = 'step-line' + (i < active ? ' done' : '');

    });

  }


  async function openCamera() {

    try {

      stream = await navigator.mediaDevices.getUserMedia({

        video: { facingMode: 'environment', width: { ideal: 1920 }, height: { ideal: 1080 } },

        audio: false

      });

      const video = document.getElementById('videoEl');

      video.srcObject = stream;

      video.style.display = 'block';

      document.getElementById('placeholder').style.display = 'none';

      document.getElementById('btnOpenCamera').style.display = 'none';

      document.getElementById('btnCapture').style.display = 'flex';

      setStatus('Camera đang hoạt động — nhấn Chụp ảnh khi sẵn sàng');

    } catch(err) {

      setStatus('Không mở được camera: ' + err.message, 'error');

    }

  }


  function capturePhoto() {

    const video = document.getElementById('videoEl');

    const canvas = document.getElementById('canvasEl');

    

    // Tối ưu hóa: Thiết lập kích thước nén tối đa là 1280px (Vẫn cực kỳ rõ nét)

    const MAX_WIDTH = 1280;

    const MAX_HEIGHT = 1280;

    let width = video.videoWidth || 1280;

    let height = video.videoHeight || 720;


    if (width > height) {

      if (width > MAX_WIDTH) {

        height *= MAX_WIDTH / width;

        width = MAX_WIDTH;

      }

    } else {

      if (height > MAX_HEIGHT) {

        width *= MAX_HEIGHT / height;

        height = MAX_HEIGHT;

      }

    }


    canvas.width  = width;

    canvas.height = height;

    

    const ctx = canvas.getContext('2d');

    ctx.drawImage(video, 0, 0, width, height);

    

    // Nén chất lượng ảnh xuống 80% (0.8) để giảm dung lượng file xuống còn ~200KB thay vì vài MB

    capturedDataUrl = canvas.toDataURL('image/jpeg', 0.8);


    // Dừng camera để tiết kiệm pin/RAM điện thoại

    if (stream) stream.getTracks().forEach(t => t.stop());

    video.style.display = 'none';


    // Hiển thị ảnh xem trước

    const img = document.getElementById('previewImg');

    img.src = capturedDataUrl;

    img.style.display = 'block';


    document.getElementById('step1Btns').style.display = 'none';

    document.getElementById('step2Btns').style.display = 'block';

    updateSteps(2);

    setStatus('Ảnh đã chụp & tối ưu dung lượng. Kiểm tra rồi nhấn Upload.');

  }


  function retakePhoto() {

    capturedDataUrl = null;

    document.getElementById('previewImg').style.display = 'none';

    document.getElementById('previewImg').src = '';

    document.getElementById('placeholder').style.display = 'flex';

    document.getElementById('step2Btns').style.display = 'none';

    document.getElementById('step1Btns').style.display = 'block';

    document.getElementById('btnOpenCamera').style.display = 'flex';

    document.getElementById('btnCapture').style.display = 'none';

    document.getElementById('btnForm').style.display = 'none';

    document.getElementById('linkBox').style.display = 'none';

    uploadedLink = null;

    updateSteps(1);

    setStatus('');

    openCamera();

  }


  async function uploadPhoto() {

    if (!capturedDataUrl) return;

    const btn = document.getElementById('btnUpload');

    btn.disabled = true;

    btn.innerHTML = '<div class="spinner"></div> Đang upload...';

    setStatus('Đang tải ảnh lên Google Drive...');


    try {

      // Đặt tên file theo định dạng thời gian thực tế

      const dateStr = new Date().toISOString().replace(/[:.]/g,'-');

      const filename = 'checkin_' + dateStr + '.jpg';


      // Gọi API bằng Fetch kết hợp CORS mượt mà, không sợ lỗi Redirect của iOS

      const response = await fetch(SCRIPT_URL, {

        method: 'POST',

        mode: 'cors',

        headers: {

          'Content-Type': 'text/plain;charset=utf-8', 

        },

        body: JSON.stringify({ 

          image: capturedDataUrl, 

          filename: filename 

        })

      });


      if (!response.ok) {

        throw new Error('Mạng không ổn định hoặc lỗi kết nối Gateway.');

      }


      const result = await response.json();


      if (result.success) {

        uploadedLink = result.url;

        setStatus('✓ Upload thành công!', 'success');

        const box = document.getElementById('linkBox');

        box.innerHTML = '🔗 <a href="' + uploadedLink + '" target="_blank">Xem ảnh trên Drive</a>';

        box.style.display = 'block';

        document.getElementById('step2Btns').style.display = 'none';

        document.getElementById('btnForm').style.display = 'flex';

        updateSteps(3);

      } else {

        throw new Error(result.error || 'Upload thất bại từ phía hệ thống Google.');

      }

    } catch(err) {

      setStatus('Lỗi: ' + err.message + ' (Hãy kiểm tra lại Apps Script đã cập nhật đúng bản triển khai mới chưa?)', 'error');

      btn.disabled = false;

      btn.innerHTML = '<svg width="18" height="18" viewBox="0 0 24 24" fill="white"><path d="M19 13h-6v6h-2v-6H5v-2h6V5h2v6h6v2z"/></svg> Thử lại';

    }

  }


  function goToForm() {

    if (!uploadedLink) return;

    const url = FORM_URL + '?' + ENTRY_ID + '=' + encodeURIComponent(uploadedLink);

    window.open(url, '_blank');

  }
